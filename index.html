-- PM Product Playbook — Supabase schema setup -- Run this once: Supabase dashboard → left sidebar → SQL Editor → New query → paste all → Run -- One row per project. owner_id is the Supabase auth user id, assigned -- when someone signs in with their email magic link. create table if not exists pm_projects ( id text primary key, owner_id uuid not null default auth.uid(), name text default '', phase_set text default 'standard', checks jsonb default '{}'::jsonb, data jsonb default '{}'::jsonb, extra_steps jsonb default '[]'::jsonb, custom_phases jsonb default null, deck_template text default 'myrec', sources jsonb default '[]'::jsonb, created_at timestamptz default now(), updated_at timestamptz default now() ); -- If you already ran an earlier version of this schema, run this too so the -- existing table picks up the columns it was missing. alter table pm_projects add column if not exists extra_steps jsonb default '[]'::jsonb; alter table pm_projects add column if not exists custom_phases jsonb default null; alter table pm_projects add column if not exists deck_template text default 'myrec'; alter table pm_projects add column if not exists sources jsonb default '[]'::jsonb; create index if not exists pm_projects_owner_idx on pm_projects (owner_id); alter table pm_projects enable row level security; -- Explicit grants: RLS is the real security boundary (below), but PostgREST -- still needs base table-level privileges before RLS policies are even -- evaluated. Being explicit here means this schema doesn't depend on -- whatever default privileges your specific project happens to have. -- authenticated = signed-in users (via magic link). anon = not signed in, -- and gets no privileges at all, since this app requires sign-in to write. grant select, insert, update, delete on pm_projects to authenticated; -- auth.uid() is the signed-in user's id once they click their magic link. -- These policies are the standard, well-tested Supabase pattern. -- Dropped and recreated each run so this script is safe to re-run anytime. drop policy if exists "owner can read own projects" on pm_projects; create policy "owner can read own projects" on pm_projects for select using (owner_id = auth.uid()); drop policy if exists "owner can insert own projects" on pm_projects; create policy "owner can insert own projects" on pm_projects for insert with check (owner_id = auth.uid()); drop policy if exists "owner can update own projects" on pm_projects; create policy "owner can update own projects" on pm_projects for update using (owner_id = auth.uid()); drop policy if exists "owner can delete own projects" on pm_projects; create policy "owner can delete own projects" on pm_projects for delete using (owner_id = auth.uid()); create or replace function set_updated_at() returns trigger as $$ begin new.updated_at = now(); return new; end; $$ language plpgsql; drop trigger if exists pm_projects_updated_at on pm_projects; create trigger pm_projects_updated_at before update on pm_projects for each row execute function set_updated_at(); -- IMPORTANT: this version uses real email-based magic-link sign-in, NOT -- anonymous auth. You do NOT need to enable "Allow anonymous sign-ins". -- Magic link / email OTP sign-in is on by default in every Supabase project. -- No further dashboard configuration is required beyond running this SQL.
